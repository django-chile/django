# Creando Usuario
```
adduser django
delsuer django
```

# entramos al usuario
```
su - django
```

usermod -aG sudo django

# Instalamos
```
sudo apt install python-virtualenv
```

# Creamos un enterno virtual
```
sudo virtualenv .venv --python=python3
```

# Activamos el enterno
```
source ~/.venv/bin/activate
```

# Instalar Django
```
pip install Django
```
